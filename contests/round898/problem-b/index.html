B. Good Kid

#include<bits/stdc++.h>
using namespace std;

const int MOD = 0;

void solve(){
	int n; cin >> n;
	vector<int> digits(n);
	int smallestDigit = INT_MAX;

	int zeroCount = 0;
	for(int i = 0; i < n; ++i){
		cin >> digits[i];
		if(digits[i] == 0) zeroCount++;
		smallestDigit = min(digits[i], smallestDigit);
	}

 	bool flag = true;
 	int ans = 1;

	for(int num : digits){
		if(num == smallestDigit && flag){
			ans *= (num+1);
			flag = !flag;
		}
		else {
			ans *= num;
		}
	}

	cout << (zeroCount > 1 ? 0 : ans) << "\n";
}

int main(){
	ios_base::sync_with_stdio(0); cin.tie(0);
	int tc; cin >> tc;
	while(tc--){
		solve();
	}
}
